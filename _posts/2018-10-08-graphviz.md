 digraph structs { 
    node [shape=record];

    struct2 [label="{ {_|_|o} |{x|x|o}| {x|_|o}}"];
    struct3 [label="{ {_|_|o} |{x|x|o}| {x|_|o}}"];
    struct4 [label="{ {_|_|o} |{x|x|o}| {x|_|o}}"];
    struct5 [label="{ {_|_|o} |{x|x|o}| {x|_|o}}"];

    struct2 -> struct3
    struct2 -> struct4
    struct3 -> struct5
 }
